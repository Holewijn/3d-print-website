model MediaImage {
  id           String   @id @default(cuid())
  filename     String   @unique          // stored on disk: <hex>.<ext>
  originalName String                    // user's original filename
  mimeType     String
  sizeBytes    Int
  width        Int?
  height       Int?
  webpFilename String?                   // 800px webp derivative
  alt          String?                   // optional accessibility label
  createdAt    DateTime @default(now())
  @@index([createdAt])
}
